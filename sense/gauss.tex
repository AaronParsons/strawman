\begin{equation}
B = e^{-(x/2\sigma^2)}
\end{equation}

\begin{equation}
B^2 = e^{-(x/2\sigma^2)} \times e^{-(x/2\sigma^2)}
\end{equation}

\begin{equation}
B^2 = e^{-(x/2\sigma^2 + x/2\sigma^2)}
\end{equation}

\begin{equation}
B^2 = e^{-(x/\sigma^2)}
\end{equation}

\begin{equation}
\sigma'^2 \equiv \frac{1}{2}\sigma^2
\end{equation}

\begin{equation}
\sigma^2 = 2\sigma'^2
\end{equation}

\begin{equation}
B^2 = e^{-(x/2\sigma'^2)}
\end{equation}

Therefore, $B^2$ is a Gaussian with new width $\sigma'$ = $\frac{1}{\sqrt{2}}\sigma$.
